def create
	#authorize! :create, Image
  @image = Image.new(image_params)
  @image.contributor = current_user
  if @image.save
    flash[:notice] = "Successfully created photo."
    redirect_to edit_interest_point_path(@image.interest_point)
  else
    render 'new'
  end
end
